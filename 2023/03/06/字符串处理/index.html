字符串处理

发表于2023-03-06|更新于2023-03-06|golang
|阅读量:

https://www.cnblogs.com/xdyixia/p/11896134.html

# strings 操作

# Contains

  • 字符串是否包含子字符串
// s是否包含 substr 的字符串
func strings.Contains(s string, substr string) bool

# Join

  • 字符串切片的元素通过某个字符串连接
// 常用逗号、分号或空格连接
func Join(a []string, sep string) string

# Index

  • 查找字符串中子字符串的位置
// 找不到则返回 -1
func Index(s, sep string) int

# Repeat

  • 复读某个字符串若干次
// count 为重复 s 字符串的次数
func Repeat(s string, count int) string

# Replace

  • 替换或删除字符串中某个子字符串
// s 为要操作的字符串，old为s中旧有的字符串，new为用来替换的新的字符串
// n为替换次数，小于0则全部替换
// 删除：用空字符串进行替换则可实现删除
func Replace(s, old, new string, n int) string

# Split

  • 分割字符串，返回字符串切片
// 常用逗号、分号或空格作为分隔符
func Split(s, sep string) []string

# Trim

  • 在字符串的头部和尾部去除指定的字符串
// 常用于去空格
// 只去头去尾，中间的忽略
func Trim(s string, cutset string) string

# Fields

  • 去除 s 字符串的空格符，并且按照空格分割返回 slice
func Fields(s string) []string

# strconv 操作

# Append

  • 将其他类型数据转换为字符串后追加
str := make([]byte, 0, 100)
str = strconv.AppendInt(str, 4567, 10) //以10进制方式追加
str = strconv.AppendBool(str, false)
str = strconv.AppendQuote(str, "abcdefg")
str = strconv.AppendQuoteRune(str, '单')
fmt.Println(string(str)) //4567false"abcdefg"'单'

# Format

  • 其他类型转换为字符串
a := strconv.FormatBool(false)
b := strconv.FormatInt(1234, 10)
c := strconv.FormatUint(12345, 10)
d := strconv.Itoa(1023)
fmt.Println(a, b, c, d) //false 1234 12345 1023

# Prase

  • 字符串转换为其他类型
func checkError(e error) {
    if e != nil {
        fmt.Println(e)
    }
}
// 字符串不一定都能转换成其他类型，所以会有 error
a, err := strconv.ParseBool("false")
checkError(err)
b, err := strconv.ParseFloat("123.23", 64)
checkError(err)
c, err := strconv.ParseInt("1234", 10, 64)
checkError(err)
d, err := strconv.ParseUint("12345", 10, 64)
checkError(err)
e, err := strconv.Atoi("1023")
checkError(err)
fmt.Println(a, b, c, d, e) //false 123.23 1234 12345 1023
